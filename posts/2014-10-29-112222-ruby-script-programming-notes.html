如何切換目錄

Dir.chdir("the/target/path/to/change/to") do
  # ... do something ...
end
# back to original path here

取得script檔案的路徑

script_path = Pathname.new(File.dirname(__FILE__))

瀏覽某個目錄下的檔案

Dir["the/target/path/to/get/files/"].each do |file| do
  # ... do something with file ...
end

可以搭配使用萬用字元找出想要的檔案

Dir["the/target/path/to/get/files/*.rb"].each do |file| do
  # ... do something with rb files ...
end
